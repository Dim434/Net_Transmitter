3 /**
4  * Wrapper for network stream functionality.
5 
6  *
7  * PHP has built in support for various types of network streams, such as HTTP and TCP sockets. One problem that arises with them is the fact that a single fread/fwrite call might not read/write all the data you intended, regardless of whether you're in blocking mode or not. While the PHP manual offers a workaround in the form of a loop with a few variables, using it every single time you want to read/write can be tedious.
8 
9 This package abstracts this away, so that when you want to get exactly N amount of bytes, you can be sure the upper levels of your app will be dealing with N bytes. Oh, and the functionality is nicely wrapped in an object (but that's just the icing on the cake).

235  /**
236  * Sets the size of the chunk.
237  *
238  * To ensure data integrity, as well as to allow for lower memory
239  * consumption, data is sent/received in chunks. This function
240  * allows you to set the size of each chunk. The default is 0xFFFFF.
241  *
242  * @param int $size The desired size of the chunk, in bytes.
243  * @param int $direction The chunk of which direction to set. Valid
244  * values are the DIRECTION_* constants.
245  *
246  * @return bool TRUE on success, FALSE on failure.
247  */
248  public function setChunk($size, $direction = self::DIRECTION_ALL)
249  {
250  $size = (int) $size;
251  if ($size <= 0) {
252  return false;
253  }
254  switch($direction) {
255  case self::DIRECTION_SEND:
256  case self::DIRECTION_RECEIVE:
257  $this->chunkSize[$direction] = $size;
258  return true;
259  case self::DIRECTION_ALL:
260  $this->chunkSize[self::DIRECTION_SEND]
261  = $this->chunkSize[self::DIRECTION_RECEIVE] = $size;
262  return true;
263  }
264  return false;
265  }

381  *
382  * Reads from the wrapped stream to receive content as a string.
383  *
384  * @param int $length The number of bytes to receive.

388  * @return string The received content.
389  */
390  public function receive($length, $what = 'data')
391  {
392  $result = '';
393  $chunkSize = $this->chunkSize[self::DIRECTION_RECEIVE];
394  while ($length > 0) {
395  while ($this->isAvailable()) {
396  $fragment = fread($this->stream, min($length, $chunkSize));
397  if ('' != $fragment) {
398  $length -= strlen($fragment);
399  $result .= $fragment;
400  continue 2;
401  } elseif (!$this->isBlocking && !(false === $fragment)) {
402  usleep(3000);
403  continue 2;
404  }
405  }
406  throw $this->createException(
407  "Failed while receiving {$what}",
408  4,
409  null,
410  $result
411  );
412  }
413  return $result;
