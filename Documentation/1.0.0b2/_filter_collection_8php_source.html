4  * Wrapper for network stream functionality.
5 
6  *
7  * PHP has built in support for various types of network streams, such as HTTP and TCP sockets. One problem that arises with them is the fact that a single fread/fwrite call might not read/write all the data you intended, regardless of whether you're in blocking mode or not. While the PHP manual offers a workaround in the form of a loop with a few variables, using it every single time you want to read/write can be tedious.
8 
9 This package abstracts this away, so that when you want to get exactly N amount of bytes, you can be sure the upper levels of your app will be dealing with N bytes. Oh, and the functionality is nicely wrapped in an object (but that's just the icing on the cake).

68  /**
69  * Inserts the filter before a position.
70  *
71  * Inserts the specified filter before a filter at a specified position. The
72  * new filter takes the specified position, while previous filters are moved
73  * forward by one.
74  *
75  * @param int $position The position before which the filter will be
76  * inserted.
77  * @param string $name The name of the filter.
78  * @param array $params An array of parameters for the filter.
79  *
80  * @return $this The collection itself.
81  */
82  public function insertBefore($position, $name, array $params = array())
83  {
84  $position = (int) $position;
85  if ($position <= 0) {
86  $this->filters = array_merge(
87  array(0 => array((string) $name, $params)),
88  $this->filters
89  );
90  return $this;
91  }
92  if ($position > count($this->filters)) {
93  return $this->append($name, $params);
94  }
95  $this->filters = array_merge(
96  array_slice($this->filters, 0, $position),
97  array(0 => array((string) $name, $params)),
98  array_slice($this->filters, $position)
99  );
100  return $this;
101  }
